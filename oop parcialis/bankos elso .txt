package BankAccout;

public class BankAccaunt {
    private String szamlaszam;
    private double egyenleg;
    public BankAccaunt() {
    }

    public BankAccaunt(String szamlaszam) {
        this.egyenleg = 0;
        this.szamlaszam = szamlaszam;

    }

    ///Getter
    public double getEgyenleg() {
        return egyenleg;
    }

    public String getSzamlaszam() {
        return szamlaszam;
    }

    ///Szeterr
    public void setEgyenleg(double egyenleg) {
        this.egyenleg = egyenleg;
    }

    public void setSzamlaszam(String szamlaszam) {
        this.szamlaszam = szamlaszam;
    }

    ///betetel
    public void beTetek(double beTet){
        if(beTet >0 )
        {
            this.egyenleg+=beTet;
        }

    }
    ///kivetel
    public boolean kiVetel(double  beTet)
    {
        if(beTet> 0 && beTet<=this.egyenleg){
      this.egyenleg-=beTet;
      return  true;
        }
        return false;
    }
    @Override
    public String toString(){
        return "Bankszamlaszam: "+this.szamlaszam+ "\nEgyenleg: "+this.egyenleg+" \n";
    }

}
